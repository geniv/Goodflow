package cz.upce.fei.ioop.soubory.file;

import java.io.File;
import java.io.FilenameFilter;


// zpracovani seznamu adresare pomoci anonymni tridy
public class SeznamAdresare2 {

  public String[] getSeznam(String cestaStr) {
    return getSeznamFiltr(cestaStr, "");
  }

  public String[] getSeznamFiltr(String cestaStr, final String filtrStr) {
    return new File(cestaStr).list(new FilenameFilter() {
      public boolean accept(File dir, String name) {
        return new File(name).getName().indexOf(filtrStr) != -1;
      }
    });
  }
}
