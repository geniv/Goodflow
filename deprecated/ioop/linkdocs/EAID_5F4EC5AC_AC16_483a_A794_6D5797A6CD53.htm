public class CekarnaTest {

  private void sout(Cekarna o) {
    System.out.println(o);
  }

  @Test
  public void testCekarna() {
    Cekarna cekarna = new Cekarna(5);

    Cekarna.Listek listek1 = new Cekarna.Listek();
    Cekarna.Listek listek2 = new Cekarna.Listek();
    Cekarna.Listek listek3 = new Cekarna.Listek();
    Cekarna.Listek listek4 = new Cekarna.Listek();
    Cekarna.Listek listek5 = new Cekarna.Listek();

    cekarna.vlozListek(listek1);
    assertEquals(listek1, cekarna.vemListek());

    try {
      assertEquals(listek1, cekarna.vemListek());
    } catch (KolekcePrazdnaException ex) {
      System.out.println("Podtečení fronty");
    }

    cekarna.vlozListek(listek1);
    cekarna.vlozListek(listek2);
    sout(cekarna);
    assertEquals(listek1, cekarna.vemListek());
    assertEquals(listek2, cekarna.vemListek());
    sout(cekarna);
    assertEquals(StavKolekce.PRAZDNY, cekarna.getStavCekarny());

    cekarna.vlozListek(listek1);
    cekarna.vlozListek(listek2);
    cekarna.vlozListek(listek3);
    cekarna.vlozListek(listek4);
    cekarna.vlozListek(listek5);
    sout(cekarna);
    assertEquals(StavKolekce.PLNY, cekarna.getStavCekarny());


    cekarna.zrus();
    assertEquals(StavKolekce.PRAZDNY, cekarna.getStavCekarny());

    sout(cekarna);
  }
}
