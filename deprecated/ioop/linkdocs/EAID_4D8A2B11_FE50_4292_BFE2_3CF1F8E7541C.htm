package zlomek;

public class Zlomek {

    protected int citatel, jmenovatel;

    public Zlomek(int citatel, int jmenovatel) {
        this.citatel = citatel;
        this.jmenovatel = jmenovatel;
    }

    public Zlomek krat(Zlomek z) {
        citatel *= z.citatel;
        jmenovatel *= z.jmenovatel;
        return this;
    }

    @Override
    public String toString() {
        return "[" + citatel + "/" + jmenovatel + "]";
    }

    public static void test() {
        Zlomek za = new Zlomek(2, 1);
        Zlomek zb = za;
        System.out.println("Zlomek pred operaci: za=" + za + ", zb=" + zb);
        zb= zb.krat(zb);
        System.out.println("Zlomek po operaci: za=" + za + ", zb=" + zb);
    }

    public static void main(String[] args) {
        test();
    }
}
