package cz.upce.fei.ioop.kolekce.fronta.object;

import cz.upce.fei.ioop.kolekce.IIteratorObject;
import cz.upce.fei.ioop.kolekce.IKolekceObject;
import cz.upce.fei.ioop.kolekce.KolekcePrazdnaException;
import cz.upce.fei.ioop.kolekce.KolekcePlnaException;
import cz.upce.fei.ioop.kolekce.StavKolekce;

public class Fronta implements IKolekceObject {

  private StavKolekce stav;
  private Object[] prvky;
  private int pocetPrvku;
  private int kapacita;

  public Fronta(int kapacita) {
    this.kapacita = kapacita;
    prvky = new Object[kapacita];
    stav = StavKolekce.PRAZDNY;
    pocetPrvku = 0;
    vynuluj();
  }

  public boolean jePrazdny() {
    return stav == StavKolekce.PRAZDNY;
  }

  public boolean jePlny() {
    return stav == StavKolekce.PLNY;
  }

  public int getPocetPrvku() {
    return pocetPrvku;
  }

  public void vlozPrvek(Object prvek) {
    if (stav == StavKolekce.PLNY) {
      stav = StavKolekce.PRETECENI;
      throw new KolekcePlnaException();
    }
    prvky[pocetPrvku] = prvek;
    pocetPrvku++;
    if (pocetPrvku == kapacita) {
      stav = StavKolekce.PLNY;
    } else {
      stav = StavKolekce.NEPRAZDNY;
    }
  }

  public Object odeberPrvek() {
    Object hodnotaOdebiranehoPrvku;

    if (stav == StavKolekce.PRAZDNY) {
      stav = StavKolekce.PODTECENI;
      throw new KolekcePrazdnaException();
    }
    if (stav == StavKolekce.PRETECENI) {
      // nepovolime pracovat s pretecenou frontou
      throw new KolekcePlnaException();
    }
    hodnotaOdebiranehoPrvku = prvky[0];
    pocetPrvku--;
    if (pocetPrvku == 0) {
      stav = StavKolekce.PRAZDNY;
    } else {
      for (int i = 0; i < pocetPrvku; i++) {
        prvky[i] = prvky[i + 1];
      }
    }
    return hodnotaOdebiranehoPrvku;
  }

  final public void vynuluj() {
    pocetPrvku = 0;
    stav = StavKolekce.PRAZDNY;
  }

  public StavKolekce getStav() {
    return stav;

  }

  public int getKapacita() {
    return prvky.length;
  }

  @Override
  public String toString() {
    String retval = "FrontaObject{" + " stav=" + stav + " pocetPrvku=" + pocetPrvku + " kapacita=" + kapacita + '}';
    if (stav == StavKolekce.NEPRAZDNY
            || stav == StavKolekce.PLNY
            || stav == StavKolekce.PRETECENI) {
      retval += "\n Obsah fronty: ";
      for (int i = 0; i < pocetPrvku; i++) {
        retval += " " + prvky[i].toString();
      }
    }
    return retval;
  }

  public IIteratorObject dejIterator() {
    return new IIteratorObject() {

      private int i = 0;

      public boolean jeDalsi() {
        return i < pocetPrvku;
      }

      public Object dalsi() {
        return prvky[i++];
      }
    };
  }
}
