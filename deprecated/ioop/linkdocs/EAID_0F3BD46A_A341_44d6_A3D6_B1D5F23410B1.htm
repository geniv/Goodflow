public class MojeException extends Exception {

    /**
     * Creates a new instance of <code>MojeException</code> without detail message.
     */
    public MojeException() {
    }

    /**
     * Constructs an instance of <code>MojeException</code> with the specified detail message.
     * @param msg the detail message.
     */
    public MojeException(String msg) {
        super(msg);
    }
}
