package cz.upce.fei.ioop.kolekce.zasobnik.object;

import cz.upce.fei.ioop.kolekce.StavKolekce;
import cz.upce.fei.ioop.kolekce.KolekcePrazdnaException;
import cz.upce.fei.ioop.kolekce.KolekcePlnaException;
import org.junit.AfterClass;
import org.junit.BeforeClass;
import org.junit.Test;
import static org.junit.Assert.*;

public class ZasobnikTest {

  TestObj to1 = new TestObj(1);
  TestObj to2 = new TestObj(2);
  TestObj to3 = new TestObj(3);

  class TestObj {

    int a;

    TestObj(int v) {
      a = v;
    }

    @Override
    public String toString() {
      return "TestObj{" + "a=" + a + '}';
    }
  }

  @Test
  public void testVytvoreniParam() {
    Zasobnik zas = new Zasobnik(10);

    assertEquals(0, zas.getPocetPrvku());
    assertEquals(10, zas.getKapacita());
    System.out.println(zas);
  }

  @Test
  public void testVlozPrvek() {
    System.out.println("\nTest VlozPrvek");
    Zasobnik zas = new Zasobnik(3);
    System.out.println(zas);

    zas.vlozPrvek(to1);
    assertEquals(1, zas.getPocetPrvku());
    System.out.println(zas);
    zas.vlozPrvek(to2);
    assertEquals(2, zas.getPocetPrvku());
    System.out.println(zas);
    zas.vlozPrvek(to3);
    assertEquals(3, zas.getPocetPrvku());
    System.out.println(zas);
    try {
      zas.vlozPrvek(4); // ctvrty prvek uz je nad ramec kapacity
      fail("Nemuze dojit k vlozeni dalsiho prvku do zasobniku");
    } catch (KolekcePlnaException ex) {
      System.out.println("Zachycena vyjimka preteceni zasobniku");
      System.out.println(zas);
      assertTrue(true);
    }
    assertEquals(3, zas.getPocetPrvku());

    try {
      Object i = zas.odeberPrvek(); // odebirame preteceny zasobnik
      fail("Nemuze odebrat z preteceneho zasobniku");
    } catch (KolekcePlnaException ex) {
      System.out.println("Zachycena vyjimka preteceni zasobniku");
      System.out.println(zas);
      assertTrue(true);
    }

    zas.vynuluj();
    System.out.println(zas);
    assertEquals(0, zas.getPocetPrvku());
  }

  @Test
  public void testOdeberPrvek() {
    System.out.println("\nTest OdeberPrvek");
    Zasobnik zas = new Zasobnik(10);

    zas.vlozPrvek(to1);
    zas.vlozPrvek(to2);
    zas.vlozPrvek(to3);

    System.out.println(zas);

    assertEquals(to3, zas.odeberPrvek());
    System.out.println(zas);
    assertEquals(to2, zas.odeberPrvek());
    System.out.println(zas);
    assertEquals(to1, zas.odeberPrvek());
    System.out.println(zas);

    try {
      zas.odeberPrvek();
      fail("Vyjimka - zasobnik je prazdny");
    } catch (KolekcePrazdnaException ex) {
      System.out.println("Zachycena vyjimka podteceni zasobniku");
      System.out.println(zas);
      assertTrue(true);
    }
    System.out.println("Nulovani obsahu zasobniku");
    zas.vynuluj();
    System.out.println(zas);
    assertEquals(0, zas.getPocetPrvku());
    assertEquals(StavKolekce.PRAZDNY, zas.getStav());
  }

  @Test
  public void testJePrazdny() {
    System.out.println("\nTest JePrazdny");
    Zasobnik zas = new Zasobnik(10);
    System.out.println(zas);

    assertEquals(true, zas.jePrazdny());

    zas.vlozPrvek(to1);
    assertEquals(false, zas.jePrazdny());
    System.out.println(zas);

    zas.vlozPrvek(to2);
    assertEquals(false, zas.jePrazdny());
    System.out.println(zas);

    zas.odeberPrvek();
    assertEquals(false, zas.jePrazdny());
    System.out.println(zas);

    zas.odeberPrvek();
    assertEquals(true, zas.jePrazdny());
    System.out.println(zas);
  }

  @Test
  public void testJePlny() {
    Zasobnik zas = new Zasobnik(3);
    assertEquals(false, zas.jePlny());

    zas.vlozPrvek(to1);
    assertEquals(false, zas.jePlny());

    zas.vlozPrvek(to2);
    assertEquals(false, zas.jePlny());

    zas.vlozPrvek(to3);
    assertEquals(true, zas.jePlny());
  }
}
