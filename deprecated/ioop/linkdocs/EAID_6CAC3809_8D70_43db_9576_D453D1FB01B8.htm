package cz.upce.fei.ioop.tridy.vnitrni.anonymni;

import cz.upce.fei.ioop.tridy.vnitrni.anonymni.Citac1;
import org.junit.Test;
import static org.junit.Assert.*;

public class Citac1Test {

  @Test(timeout = 10000)
  public void testMujTimer() {
    int c = 0;
    int citac = 0;
    Citac1 timer = new Citac1(1000);
    do {
      c = timer.getCitac();
      if (c != citac) {
        System.out.println("Citac=" + c);
        citac = c;
      }
    } while (c < 10);
  }
}
