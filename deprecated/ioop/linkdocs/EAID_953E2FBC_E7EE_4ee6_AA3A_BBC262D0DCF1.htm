package cz.upce.fei.ioop.soubory.file;

import java.io.File;
import java.io.FilenameFilter;

public class SeznamAdresare1 {

  // vnitrni trida
  class DirFilter implements FilenameFilter {
    String filtr;
    public DirFilter(String filtr) {
      this.filtr = filtr;
    }
    // metoda pro zpetne volani tzv. callback
    public boolean accept(File dir, String name) {
      String f = new File(name).getName();
      return f.indexOf(filtr) != -1;
    }
  }

  // doda cely seznam v adresari
  public String[] getSeznam(String cestaStr) {
    File cesta = new File(cestaStr);
    String[] seznam = cesta.list();
    return seznam;
  }

  // doda filtrovany seznam adresare
  public String[] getSeznam(String cestaStr, String filtrStr) {
    File cesta = new File(cestaStr);
    String[] seznam = cesta.list(new DirFilter(filtrStr));
    return seznam;
  }
}
