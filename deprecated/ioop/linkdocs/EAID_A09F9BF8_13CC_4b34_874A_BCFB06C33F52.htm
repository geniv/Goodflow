package cz.upce.fei.ioop.kolekce;

public enum StavKolekce {

  NEZNAMY(0), PRAZDNY(1), NEPRAZDNY(2), PLNY(3), PRETECENI(4), PODTECENI(5);
  private final int value;

  StavKolekce(int value) {
    this.value = value;
  }

  @Override
  public String toString() {
    switch (this) {
      case NEZNAMY:
        return "Neznámý";
      case PRAZDNY:
        return "Prázdný";
      case NEPRAZDNY:
        return "Neprázdný";
      case PLNY:
        return "Plný";
      case PRETECENI:
        return "Přetečení";
      case PODTECENI:
        return "Podtečení";
    }
    return "StavKolekce{" + "value=" + value + '}';
  }
}
