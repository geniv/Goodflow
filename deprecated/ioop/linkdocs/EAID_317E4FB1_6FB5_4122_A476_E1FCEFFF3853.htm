package koule;

import java.util.logging.Level;
import java.util.logging.Logger;
import org.junit.AfterClass;
import org.junit.BeforeClass;
import org.junit.Test;
import static org.junit.Assert.*;

/**
 * Test třídy Koule
 *
 * @author kasi0004
 */
public class KouleTest {

    Koule k;

    public KouleTest() {
        k = new Koule();
    }

    /**
     * Test na zmenu polomeru koule v povolených mezích
     */
    @Test
    public void testPolomer() throws MojeException {

            double r = 0.0;
            k.setPolomer(r);
            assertEquals(r, k.getPolomer(), 0.001);

            r = .001;
            k.setPolomer(r);
            assertEquals(r, k.getPolomer(), 0.001);

            r = 1.0;
            k.setPolomer(r);
            assertEquals(r, k.getPolomer(), 0.001);
           
            r = 10.0;
            k.setPolomer(r);
            assertEquals(r, k.getPolomer(), 0.001);


    }

    /**
     * Test na vyhozeni výjimky, která musí nastat tj.
     *  hodnota poloměru mimo povolený rzsah
     */
    @Test(expected=MojeException.class)
    public void testPolomerNaVyjimku1() throws MojeException {
            k.setPolomer(-0.001);
    }
   
    @Test(expected=MojeException.class)
    public void testPolomerNaVyjimku2() throws MojeException {
            k.setPolomer(10.001);
    }
   
    /**
     * Test na nastavení barvy .
     */
    @Test
    public void testBarva() {
        assertEquals(MojeBarva.CERNA, k.getBarva());
        k.setBarva(MojeBarva.BILA);
        assertEquals(MojeBarva.BILA, k.getBarva());
        k.setBarva(MojeBarva.CERVENA);
        assertEquals(MojeBarva.CERVENA, k.getBarva());
        k.setBarva(MojeBarva.MODRA);
        assertEquals(MojeBarva.MODRA, k.getBarva());
        k.setBarva(MojeBarva.ZELENA);
        assertEquals(MojeBarva.ZELENA, k.getBarva());
    }

    /**
     * Test na hmotnost Koule.
     */
    @Test
    public void testHmotnost()  {
        try {
            k.setPolomer(1.0);
            k.setMernaHmotnost(1.0);
        } catch (MojeException ex) {
            Logger.getLogger(KouleTest.class.getName()).log(Level.SEVERE, null, ex);
        }
        assertEquals(4188.79, k.getHmotnost(), 0.001);
    }

     *  hodnota merne hmotnosti je mimo povolený rzsah
     */

            k.setMernaHmotnost(-0.001);

    public void testMernaHmotnost() throws MojeException {
