package Koule;

import org.junit.Test;
import static org.junit.Assert.*;

public class KouleTest {

    final double r = 10.0;
    Koule koule;

    public KouleTest() {
        koule = new Koule();
        koule.setPolomer(r);
    }
  
    @Test
    public void  testObjem( ) {
        koule.setPolomer(r);
        double result = koule.objem();
        double expResult = (4.0 / 3.0) * Math.PI * Math.pow(r, 3);
        assertEquals(expResult, result, 0.1);
    }

    @Test
    public void testSetPolomer() {
        koule.setPolomer(2 * r);
        double result = koule.getPolomer();
        double expResult = 2 * r;
        assertEquals(expResult, result, 0.1);
    }

    @Test
    public void testGetPolomer() {
        Koule k = new Koule();
        k.setPolomer(2 * r);
        double result = k.getPolomer();
        double expResult = 2 * r;
        assertEquals(expResult, result, 0.1);
    }
}
