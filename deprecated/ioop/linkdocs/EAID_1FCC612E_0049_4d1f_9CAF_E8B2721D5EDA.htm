package cz.upce.fei.ioop.zasobnik;

import cz.upce.fei.ioop.zasobnik.celacisla.StavKolekce;
import cz.upce.fei.ioop.zasobnik.vyjimky.ZasobnikPrazdnyException;
import cz.upce.fei.ioop.zasobnik.vyjimky.ZasobnikPlnyException;
import cz.upce.fei.ioop.zasobnik.celacisla.ZasobnikInt;
import org.junit.Test;
import static org.junit.Assert.*;

public class ZasobnikIntTest {

  @Test
  public void testVytvoreniParam() {
    ZasobnikInt zas = new ZasobnikInt(10);

    assertEquals(0, zas.getPocetPrvku());
    assertEquals(10, zas.getKapacitaZasobniku());
    System.out.println(zas);
  }

  @Test
  public void testVlozPrvek() {
    System.out.println("\nTest VlozPrvek");
    ZasobnikInt zas = new ZasobnikInt(3);
    System.out.println(zas);

    zas.vlozPrvek(5);
    assertEquals(1, zas.getPocetPrvku());
    System.out.println(zas);
    zas.vlozPrvek(6);
    assertEquals(2, zas.getPocetPrvku());
    System.out.println(zas);
    zas.vlozPrvek(7);
    assertEquals(3, zas.getPocetPrvku());
    System.out.println(zas);
    try {
      zas.vlozPrvek(4); // ctvrty prvek uz je nad ramec kapacity
      fail("Nemuze dojit k vlozeni dalsiho prvku do zasobniku");
    } catch (ZasobnikPlnyException ex) {
      System.out.println("Zachycena vyjimka preteceni zasobniku");
      System.out.println(zas);
      assertTrue(true);
    }
    assertEquals(3, zas.getPocetPrvku());

   try {
      int i= zas.odeberPrvek(); // odebirame preteceny zasobnik
      fail("Nemuze odebrat z preteceneho zasobniku");
    } catch (ZasobnikPlnyException ex) {
      System.out.println("Zachycena vyjimka preteceni zasobniku");
      System.out.println(zas);
      assertTrue(true);
    }

    zas.vynuluj();
    System.out.println(zas);
    assertEquals(0, zas.getPocetPrvku());
  }

  @Test
  public void testOdeberPrvek() {
    System.out.println("\nTest OdeberPrvek");
    ZasobnikInt zas = new ZasobnikInt(10);

    zas.vlozPrvek(5);
    zas.vlozPrvek(6);
    zas.vlozPrvek(7);

    System.out.println(zas);

    assertEquals(7, zas.odeberPrvek());
    System.out.println(zas);
    assertEquals(6, zas.odeberPrvek());
    System.out.println(zas);
    assertEquals(5, zas.odeberPrvek());
    System.out.println(zas);

    try {
      zas.odeberPrvek();
      fail("Vyjimka - zasobnik je prazdny");
    } catch (ZasobnikPrazdnyException ex) {
      System.out.println("Zachycena vyjimka podteceni zasobniku");
      System.out.println(zas);
      assertTrue(true);
    }
    System.out.println("Nulovani obsahu zasobniku");
    zas.vynuluj();
    System.out.println(zas);
    assertEquals(0, zas.getPocetPrvku());
    assertEquals(StavKolekce.PRAZDNY,zas.getStav());
  }

  @Test
  public void testJePrazdny() {
    System.out.println("\nTest JePrazdny");
    ZasobnikInt zas = new ZasobnikInt(10);
    System.out.println(zas);

    assertEquals(true, zas.jePrazdny());

    zas.vlozPrvek(5);
    assertEquals(false, zas.jePrazdny());
    System.out.println(zas);

    zas.vlozPrvek(6);
    assertEquals(false, zas.jePrazdny());
    System.out.println(zas);

    zas.odeberPrvek();
    assertEquals(false, zas.jePrazdny());
    System.out.println(zas);

    zas.odeberPrvek();
    assertEquals(true, zas.jePrazdny());
    System.out.println(zas);
  }

  @Test
  public void testJePlny() {
    ZasobnikInt zas = new ZasobnikInt(3);
    assertEquals(false, zas.jePlny());

    zas.vlozPrvek(5);
    assertEquals(false, zas.jePlny());

    zas.vlozPrvek(6);
    assertEquals(false, zas.jePlny());

    zas.vlozPrvek(5);
    assertEquals(true, zas.jePlny());
  }
}
