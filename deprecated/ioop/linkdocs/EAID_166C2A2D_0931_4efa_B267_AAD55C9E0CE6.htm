package cz.upce.fei.ioop.tridy.vnitrni.anonymni;

/**
 * Rozhraní ICitace definuje ovládání seznamu čítačů.
 * <p>
 * U každého čítače lze nastavit na periodu, kdy bude zvýšena jeho
 * hodnota o 1.
 * </p>
 *
 * <p><b>Poznámky k implementaci:</b><br>
 * <ul>
 *   <li>
 *      Toto rozhraní předpokládá, že implementace seznamu čítačů bude
 *      na poli, tj. že první čítač bude na indexu 0.
 *   </li>
 *   <li>
 *      Kapacita seznamu by měla být nastavena konstruktorem,
 *      nejlépe jako parametr.
 *   </li>
 * </ul>
 * @author kasi0004
 */
public interface ICitace {

  /**
   * Získání aktuální hodnoty čítače
   *
   * @param index  místo čítače pod kterým byl vložen do tabulky čítačů
   * @return vrací aktuální hodnotu čítače
   */
  long getHodnotaCitace(int index);

  /**
   * Přidání čítače do seznamu čítačů
   *
   * @param perioda doba v milisekundach za kterou se zvýší hodnota
   * čítače o 1.
   * @param index místo čítače v seznamu
   */
  void addNovyCitac(long perioda, int index);

  void start();

  void stop();

  void start(int index);

  void stop(int index);

  /**
   * Nastavení hodnot všech čítačů na nulu.
   */
  void vynulujVsechnyCitace();

  /**
   * Vynulování hodnoty zvoleného čítače
   *
   * @param index místo čítače v seznamu
   */
  void vynulujCitac(int index);

  /**
   *  Zruší všechny čítače
   */
  void zrusCitace();

  /**
   * Zruší zvoleny čítač
   * @param index
   */
  void zrusCitac(int index);

  /**
   * Počet čítačů v seznamu.
   *
   * <p><b>Poznámka k implementaci:</b><br>
   * Protože seznam bude tvořit tabulka s přímým přístupem pomocí indexu
   * a protože lze libovolný čítač zrušit, nemusí nejvyšší index být
   * roven počtu čítačů v tabulce.
   * </p>
   *
   * @return vratí počet čítačů
   */
  int pocetCitacu();

  /**
   * Kapacita seznamu čítačů, tj. maximální počet čítačů,
   * které lze umístit do seznamu.
   *
   * @return vrátí maximální počet čítačů
   */
  int kapacita();
}
