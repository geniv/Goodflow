package cz.upce.fei.ioop.zasobnik.celacisla;

import cz.upce.fei.ioop.zasobnik.vyjimky.ZasobnikPlnyException;
import cz.upce.fei.ioop.zasobnik.vyjimky.ZasobnikPrazdnyException;

public class ZasobnikInt {

  ... ... stav;
  ... ... prvky;
  ... ... pocetPrvku;
  ... ... kapacita;

  public ZasobnikInt(int kapacita) {
    this.kapacita = kapacita;
    prvky = ... ...[...];
    stav = ... .PRAZDNY;
    pocetPrvku = 0;
  }

  public boolean jePrazdny() {
    ... stav == StavKolekce.PRAZDNY;
  }

  public boolean jePlny() {
    ... stav == StavKolekce.PLNY;
  }

  public int getPocetPrvku() {
    ... pocetPrvku;
  }

  public void vlozPrvek(int hodnota) {
    if (stav == .....PLNY) {
      stav = ....PRETECENI;
      throw new ZasobnikPlnyException();
    }
    prvky[...++] = hodnota;
    if (pocetPrvku == kapacita) {
      stav = ....PLNY;
    } else {
      stav = StavKolekce.NEPRAZDNY;
    }
  }

  public int odeberPrvek() {
    int hodnotaOdebiranehoPrvku;

    if (stav == ....PRAZDNY) {
      stav = ....PODTECENI;
      throw new ZasobnikPrazdnyException();
    }
    if (stav == ....PRETECENI) {
      // nepovolime pracovat s pretecenym zasobnikem
      throw new ZasobnikPlnyException();
    }
    hodnotaOdebiranehoPrvku = prvky[--...];
    if (pocetPrvku == 0) {
      stav = ....PRAZDNY;
    }
    return hodnotaOdebiranehoPrvku;
  }

  public void vynuluj() {
    pocetPrvku = 0;
    stav = ....PRAZDNY;
  }

  public StavKolekce getStav() {
    return stav;
  }

  public int getKapacitaZasobniku() {
    return prvky.length;
  }

  @Override
  public String toString() {
    String retval = "ZasobnikInt{" + " stav=" + stav + " pocetPrvku=" + pocetPrvku + " kapacita=" + kapacita + '}';
    if (stav == StavKolekce.NEPRAZDNY || stav == StavKolekce.PRETECENI) {
      retval += "\n Obsah zasobniku: ";
      for (int i = 0; i < pocetPrvku; i++) {
        retval += " " + prvky[i];
      }
    }
    return retval;
  }}
