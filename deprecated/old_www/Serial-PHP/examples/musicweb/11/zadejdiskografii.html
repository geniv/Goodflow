  function pisenid ($nazev)
  {
    $vysledek=mysql_query("select id from pisne where nazev='".$nazev."'", $GLOBALS["link"]);
    if (mysql_num_rows($vysledek)==0)
      return 0;
    else {
      $radek = mysql_fetch_array($vysledek);
      return $radek["id"];
    }
  }

    }
    echo $problem;
    return (boolean)($problem==="");
  }
