Kompletní BNF zápis pøíkazu SELECT

dotazový_výraz ::= { dotazový_term | dotazový_výraz { UNION | EXCEPT } [ ALL ] 
    [ corresponding ] dotazový_term } [ tøídìní ] [ výbìr ]
dotazový_term ::= specifikace_dotazu | dotazový_term INTERSECT [ ALL ] [ corresponding ] 
    specifikace_dotazu
corresponding ::= CORRESPONDING [ BY ( jméno_sloupce {, jméno_sloupce }… )]

specifikace_dotazu ::= SELECT [ ALL | DISTINCT ] výraz [ pøejmenování ] { , výraz [ pøejmenování ] }… 
FROM zobecnìná tabulka { , zobecnìná tabulka }… 
[ WHERE podmínka_where ] 
[ GROUP BY výraz_group { , výraz_group }… ]
[ HAVING podmínka_having ]
pøejmenování ::= AS jméno

zobecnìná tabulka ::= { identifikátor_tabulky [ specifikace_indexu ] 
    | identifikátor_uloženého_dotazu | systémový_dotaz | ( dotazový_výraz ) | join }
    [ pøejmenování ]
specifikace_indexu ::= INDEX jméno_indexu
join ::= zobecnìná_tabulka druh_joinu zobecnìná_tabulka [ specifikace_joinu ]
druh_joinu ::= CROSS JOIN | [ NATURAL ] [ INNER | { LEFT | RIGHT | FULL } [ OUTER ] ] JOIN 
specifikace_joinu ::= ON podmínka | USING ( název_sloupce { , název_sloupce }…)
pøejmenování ::= [ AS ] jméno [( jméno_sloupce { , jméno_sloupce }…)]

tøídìní ::= ORDER BY výraz [ ASC | DESC ] {, výraz [ASC | DESC ] }… 
výbìr ::= LIMIT [ offset , ] count
