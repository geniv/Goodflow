Problém producenta a konzumenta

  • jeden z klasických synchronizačních problémů, který demonstruje problémy paralelního programování
  • producent produkuje nějaká data a vkládá je do sdílené paměti
  • konzument vybírá data ze sdílené paměti
[editovat]

Řešení pomocí Sleep() a Wakeup()

define N 100                           // Počet míst v bufferu
int count = 0;                         // Počet položek, které jsou aktuálně v bufferu

void producer(void) {
  int item;
  while(TRUE){
    item = produce_item();
    if (count == N) sleep();           // Pokud je buffer plný, zablokujeme se
    insert_item(item);
    count = count + 1;
    if (count == 1) wakeup(consumer);  // Pokud byl předtím buffer prázdný, uvědom konzumenta, že je co zpracovávat
  }
}

void consumer(void){
  int item;
  while (TRUE){
    if (count == 0)                    // Kritické místo pro přepnutí kontextu
      sleep();                         // Pokud je buffer prázdný, uspi se
    remove_item(&item)
    count = count – 1;
    if (count == N - 1)                // Pokud byl buffer před odebráním položky plný
      wakeup(producer);
    consume_item(&item);
  }
}
[editovat]

Řešení pomocí semaforů 

#define N 100                          // Počet míst v bufferu
semaphore mutex = 1;                   // Semafor hlídající přístup do kritické sekce
semaphore empty = N;                   // Počet volných míst v bufferu
semaphore full = 0;                    // Počet obsazených míst v bufferu

void producer(void) {
  itemtype item;
  while (TRUE) {
    produce_item(&item);
    down(&empty);
    down(&mutex);                      // Vstup do kritické sekce
    enter_item(item);
    up(&mutex);                        // Opusť kritickou sekci
    up(&full);
  }
}

void consumer(void) {
  itemtype item;
    while (TRUE) {
    down(&full);
    down(&mutex);                      // Vstup do kritické sekce
    remove_item(&item);
    up(&mutex);                        // Opusť kritickou sekci
    up(&empty);
    consume_item(item);
  }
}
[editovat]

Řešení pomocí zasílání zpráv 

#define N 100                          // Počet míst v bufferu

void producer(void) {
  int item;
  message m, e;
  while (TRUE) {
    item = produce_item();
    receive(consumer,&e);              // Čekej na zprávu "v bufferu je volno"
    build_message(&m,item);            // Vyrob zprávu
    send(consumer,&m);                 // Odešli zprávu konzumentovi
  }
}

void consumer(void) {
  int item, i;
  message m, e;
  for (i = 0; i < N; i++)
    send(producer, &e);                // Pošli N zpráv "v bufferu je volno"
  while (TRUE) {
    receive(producer, &m);             // Přijmi zprávu
    item = extract_item(&m)            // Přečti prvek ze zprávy
    send(producer,&e);                 // Uvolnili jsme místo v bufferu - pošli zprávu "v bufferu je volno"
    consume_item(item); 
  }
}

Večeřící filosofové

  • model procesů, které souteží o výlučný přístup k omezenému počtu sdílených prostředků
  • N filosofů sedí kolem kulatého stolu
  • mezi každými dvěmi filosofy je jedna vidlička (celkem N vidliček)
  • každý z filosofů buď přemýšlí, nebo jí
  • k jídlu potřebuje současně pravou i levou vidličku
[editovat]

Naivní řešení

  • všichni filosofové vezmou levou vidličku současně a potom budou čekat, až se uvolní pravá vidlička
  • snadno vede k uváznutí (budou čekat do nekonečna)
  • vylepšení: pokud není pravá vidlička k dispozici, filosof vrátí již alokovanou levou vidličku zpět na stůl a pokus o jídlo zopakuje později
  • vylepšení řešení může také uváznout:
    • všichni filosofové vezmou levou vidličku
    • pak se pokusí vzít pravou vidličku, ale protože není volná, vrátí levou vidličku zpět
    • toto budou do nekonečna opakovat => stárnutí (sice nečekají, ale donekonečna provádí to samé a nikdy se jim to nepovede dokončit)
[editovat]

Řešení pomocí semaforů 

#define N 5
#define LEFT ((i+N-1) % N)
#define RIGHT ((i+1) % N)

enum stat(thinking, hungry, eating);
enum stat state[N];
semaphore mutex = 1;
semaphore s[N];                        // Na počátku nastaveny na 0

void philosopher(int i){
  while (TRUE) {
    think();
    take_forks(i);
    eat();
    put_forks(i); 
  }
}

void take_forks(int i) {
  down(&mutex);
  state[i] = hungry;
  test(i);
  up(&mutex);
  down(&s[i]);
}

void put_forks(int i) {
  down(&mutex);
  state[i] = thinking;
  test(LEFT);
  test(RIGHT);
  up(&mutex);
}

void test(int i){
  if (state[i] == hungry &&  state[LEFT] != eating && state[RIGHT] != eating)  {
    state[i] = eating;
    up(&s[i]);
  }
}

Čtenáři a písaři

  • model procesů, které přistupují do společné databáze
  • více čtenářů může číst současně data, ale pouze pokud žádný písař nemodifikuje data v databázi
  • pouze jeden písař může modifikovat data v databázi v jednom okamžiku
[editovat]

Řešení pomocí semaforů 

int rc = 0;                            // Počítadlo čtenářů
semaphore mutex = 1;
semaphore db = 1;                      // Přístup do databáze

void reader(void){
  while(TRUE){
    down(&mutex);
    rc = rc + 1;
    if (rc == 1) down(&db);
    up(&mutex);
    read_data_base();                  // Kritická sekce
    down(&mutex);
    rc = rc - 1;
    if (rc == 0) up(&db);
    up(&mutex);
    use_data_read();                   // Nekritická sekce
  }
}

void writer(void){
  while(TRUE) {
    think_up_data();
    down(&db);
    write_data_data();
    up(&db);
  }
}

Spící holiči

  • v holičství je N holičů (barber), N holících křesel (barber chair) a M čekacích křesel (waiting chair) pro zákazníky
  • pokud není žádný zákazník v holičství, holič sedne do holícího křesla a usne
  • pokud přijde zákazník, potom:
    • pokud nějaký holič spí, tak ho zákazník probudí a nechá se ostříhat
    • jinak si sedne do křesla a čeká (pokud nějaké je volné)
    • jinak opustí holičství
[editovat]

Řešení pomocí semaforů 

#define CHAIRS 5
int waiting = 0;                       // Čekající zákazníci (ale ne ti, co jsou stříhaní)
semaphore mutex = 1;                   // Pro vzájemné vyloučení
semaphore customers = 0;               // Počet zákazníků čekajících na obsloužení
semaphore barbers = 0;                 // Počet holičů, kteří čekají na zákazníka

void Barber(void){
  while (TRUE) {
    down(&customers);                  // Pokud je nějaký zákazník, tak se sníží počet v semaforu customers, jinak se    
                                       // holič uspí
    down(&mutex)                       // Vstup do kritické sekce
    waiting = waiting - 1;             // Čeká o jednoho zákazníka méně
    up(&barbers);                      // Pokud spí nějaký jiný holič, tak ho probudí, jinak se zvýší počet čekajících  
                                       // holičů o 1
    up(&mutex);                        // Konec kritické sekce
    cut_hair();                        // Ostříhá vlasy
  }
}

void Customer(void){
  down(&mutex);                        // Vstup do kritické sekce
  if (waiting < CHAIRS) {              // Pokud je počet čekajících zákazníků menší nez počet křesel na čekání
    waiting = waiting + 1;             // Počet čekajících zákazníků se zvýší o 1
    up(&customes);                     // Pokud nějaký zákazník čeká na ostříhání, tak ho vzbudí, jinak se pocet 
                                       // zákazníků čekajících na ostříhání zvýší o 1
    up(&mutex);                        // Konec kritické sekce
    down(&barbers);                    // Pokud je nějaký holič volný, tak se zákazník nechá ostříhat, jinak holič usne
    get_haircut();                     // Nechá se ostříhat
  } else {
    up(&mutex);                        // Konec kritické sekce
  }
}
